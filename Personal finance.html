def categorize(purchase):
    if "CLIPPER" in purchase:
        return "transportation"
    if "CAFE" in purchase or "COFFEE" in purchase or "STRADA" in purchase or "NORMAL" in purchase or "BEANS" in purchase:
        return "coffee"
    if "TRADER" in purchase:
        return "trader joes"
    if "BRANDY" in purchase or "FREE PEOPLE" in purchase or "URBAN" in purchase or "ZARA" in purchase or "HAT" in purchase:
        return "clothes"
    if "Yoga" in purchase:
        return "yoga"
    if "WAX" in purchase or "ax" in purchase or "PARLOR" in purchase:
        return "wax or threading"
    if "CINEMA" in purchase:
        return "movie"
    if "CHEESE" in purchase or "SWEETGREEN" in purchase or "POKE" in purchase or "SQ" in purchase or "CKE" in purchase:
        return "food"
    if "ARCO" in purchase or "76" in purchase or "PARK" in purchase:
        return "gas/parking"
    if "CHARGE" in purchase:
        return "interest charge"
    if "FEE" in purchase:
        return "fee"
    if "GODADDY" in purchase:
        return "domain"
    if "ULTA" in purchase or "WHOLE" in purchase or "TARGET" in purchase or "FOOD CO" in purchase:
        return "misc"
    return str(purchase)
